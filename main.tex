\newcount\fileisnonempty
\makeatletter
\newcommand{\checkemptyfile}[1]{%
    \openin\testfile=#1
    \fileisnonempty=0
    \read\testfile to \dummyline
    \ifeof\testfile
    \else
        \fileisnonempty=1
    \fi
    \closein\testfile
}
\makeatother
